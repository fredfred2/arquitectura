@RequestScoped

    private String licenseKey;

    }

    public void setLicenseKey(String licenseKey) {
        this.licenseKey = licenseKey;
    }

    public String getLicenseKey() {
        return licenseKey;
    }

    public String validateLicense() {
        return "success";
    }
}
